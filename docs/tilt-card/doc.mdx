## Usage

```tsx
import { TiltCard } from "@/components/tilt-card";
```
```tsx
<TiltCard tiltLimit={15} scale={1.05} perspective={1200}>
  <div className="w-80 h-48 rounded-2xl bg-neutral-100 dark:bg-neutral-900 p-6">
    Your content here 
  </div>
</TiltCard>
```
